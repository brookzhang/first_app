git checkout master    git co
git checkout -b modify-readme       new branch
git status
git branch
git branch -d modify-readme
git branch -D modify-readme     delte branch even though we haven't merged in the changes
git merge modify-readme
git commit -m "message"         commit modified one file 
git commit -a -m "message"      commit all modified files ,not include new files
git add .                       add all include new file
git mv README README.markdown   rename file
git log
